class Persona {
  nombre = 'Juan';
  duerme(): Promise<string> {
    return new Promise((resolve) => {
      resolve(`despierta ${this.nombre}`);
    });
  }
}
const p = new Persona();
p.duerme().then(console.log); // despierta Juan
