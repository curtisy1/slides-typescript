## Tipado - Type Guards

¿Cómo podemos comprobar que una variable es de un tipo **en ejecución**?
```ts
interface Perro {
    name: string;
    ladra: () => string;
}
interface Gato {
    name: string;
    maulla: () => string;
}
function getAnimal(nombre: string) : Perro | Gato {
    if (nombre === 'Larry') {
        return <Gato>{ nombre: 'Larry', maulla: () => 'miau!'};
    } else {
        return <Perro>{ nombre, ladra: () => 'guau!'};
    }
}

```
